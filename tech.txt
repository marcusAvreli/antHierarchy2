CREATE TABLE IF NOT EXISTS hr_source(

id varchar(32) not null
,created BIGINT not null
,modified BIGINT
,company_code VARCHAR(3)
,company_name VARCHAR(450)
,contract_code INT
,cost_center VARCHAR(10)
,email VARCHAR(255)
,firstName VARCHAR(450)
,image VARCHAR(512)
,job_key VARCHAR(10)
,job_name VARCHAR(200)
,lastName VARCHAR(450)
,name VARCHAR(255)
,org_unit_code VARCHAR(10)
,phoneNumber VARCHAR(20)
,position_code VARCHAR(10)
,title VARCHAR(255)
,manager_id VARCHAR(32)
,branch_code VARCHAR(10)
,beranch_name  VARCHAR(450)
,parent_company_code VARCHAR(3)
,parent_company_name VARCHAR(450)
  ,PRIMARY KEY  (id)
)




 SELECT
	cemp.id,
	cemp.created,
	cemp.modified,      
	cemp.company_code,
	c.company_name AS company_name,       
cemp.contract_code
,cemp.cost_center
,cemp.email
,cemp.firstName
,cemp.image
,cemp.job_key
,cemp.job_name
,cemp.lastName
,cemp.name
,cemp.org_unit_code
,cemp.phoneNumber
,cemp.position_code
,cemp.title
 ,cemp.manager_id,

       
      
        br.branch_code branch_code,
        br.branch_name branch_name,

         CASE WHEN pc.company_code IS NOT NULL THEN pc.company_code ELSE '' END as company_parent_id ,
        CASE WHEN pc.company_name IS NOT NULL THEN pc.company_name ELSE '' END as company_parent_name
        
    FROM employee cemp
  
    LEFT JOIN company c ON c.company_code = cemp.company_code
    LEFT JOIN company pc ON pc.id = c.parent_id
left join branch br on br.id = cemp.branch_id
